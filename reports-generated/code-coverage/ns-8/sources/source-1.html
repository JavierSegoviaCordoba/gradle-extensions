 public inline fun <reified T : Task> TaskContainer.namedLazily(
     name: String,
     action: Action<in T>? = null,
 ): TaskCollection<T> {
     val collection: TaskCollection<T> = withType<T>().matching { it.name == name }
     if (action != null) collection.configureEach(action)
     return collection
 }
