                 withDebug(withDebug)

                 if (isolated) {
                     val isolatedDir = isolatedTempDir.resolve(name).apply(File::mkdirs)
                     withTestKitDir(isolatedDir)
                 }
                 withProjectDir(projectDir)
             },
         )
     }
